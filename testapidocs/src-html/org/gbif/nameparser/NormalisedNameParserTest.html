001package org.gbif.nameparser;
002
003import org.gbif.api.model.checklistbank.ParsedName;
004
005import java.util.regex.Pattern;
006
007import org.junit.Test;
008
009import static org.junit.Assert.assertFalse;
010import static org.junit.Assert.assertTrue;
011
012public class NormalisedNameParserTest {
013
014  @Test
015  public void testEpithetPattern() throws Exception {
016    Pattern epi = Pattern.compile("^"+NormalisedNameParser.EPHITHET+"$");
017    assertTrue(epi.matcher("alba").find());
018    assertTrue(epi.matcher("biovas").find());
019    assertTrue(epi.matcher("serovat").find());
020    assertTrue(epi.matcher("novo-zelandia").find());
021    assertTrue(epi.matcher("novae zelandia").find());
022
023    assertFalse(epi.matcher("").find());
024    assertFalse(epi.matcher("a").find());
025    assertFalse(epi.matcher("serovar").find());
026    assertFalse(epi.matcher("genotype").find());
027    assertFalse(epi.matcher("agamovar").find());
028    assertFalse(epi.matcher("cultivar").find());
029    assertFalse(epi.matcher("serotype").find());
030    assertFalse(epi.matcher("form").find());
031    assertFalse(epi.matcher("nvar").find());
032    assertFalse(epi.matcher("cytoform").find());
033    assertFalse(epi.matcher("chemoform").find());
034    assertFalse(epi.matcher("form").find());
035  }
036
037  @Test
038  public void timeoutLongNames() throws Exception {
039    final int timeout = 50;
040    NormalisedNameParser parser = new NormalisedNameParser(timeout);
041
042    String name = "Equicapillimyces hongkongensis S.S.Y. Wong, A.H.Y. Ngan, Riggs, J.L.L. Teng, G.K.Y. Choi, R.W.S. Poon, J.J.Y. Hui, F.J. Low, Luk &";
043
044    ParsedName n = new ParsedName();
045    final long pStart = System.currentTimeMillis();
046    assertFalse("No timeout happening for long running parsing", parser.parseNormalisedName(n, name, null));
047    final long duration = System.currentTimeMillis() - pStart;
048
049    // the duration is the timeout PLUS some initialization overhead thats why we add 50ms to it
050    assertTrue("No timeout happening for long running parsing", duration < 50 + timeout);
051  }
052
053}
