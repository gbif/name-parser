615   @Test
616   public void testNameFile() throws Exception {
617     LOG.info("\n\nSTARTING FULL PARSER\n");
618     Reader reader = FileUtils.getInputStreamReader(streamUtils.classpathStream(NAMES_TEST_FILE));
619     LineIterator iter = new LineIterator(reader);
620 
621     int parseFails = 0;
622     int parseNoAuthors = 0;
623     int parseErrors = 0;
624     int cnt = 0;
625     int lineNum = 0;
626     long start = System.currentTimeMillis();
627 
628     while (iter.hasNext()) {
629       lineNum++;
630       String line = iter.nextLine();
631       if (lineNum == 1 || line == null || line.startsWith("#") || line.trim().isEmpty()) {
632         continue;
633       }
634       cnt++;
635       String[] cols = FileUtils.TAB_DELIMITED.split(line);
636       String name = cols[0];
637       if (cols.length < 14) {
638         LOG.warn("Too short line in names test file:");
639         LOG.warn(line);
640       }
641       ParsedName expected = buildParsedNameFromTabRow(cols);
642       ParsedName n;
643       try {
644         n = parser.parse(name);
645         if (!n.isAuthorsParsed()) {
646           parseNoAuthors++;
647         }
648         // copy scientific name, we dont wanna compare it, it might be slightly different
649         expected.setScientificName(n.getScientificName());
650         if (!n.equals(expected)) {
651           parseErrors++;
652           LOG.warn("WRONG\t " + name + "\tEXPECTED: " + expected + "\tPARSED: " + n);
653         }
654       } catch (Exception e) {
655         parseFails++;
656         LOG.warn("FAIL\t " + name + "\tEXPECTED: " + expected);
657       }
658     }
659     long end = System.currentTimeMillis();
660     int successfulParses = cnt - parseFails - parseErrors;
661     LOG.info("\n\nNames tested: " + cnt);
662     LOG.info("Names success: " + successfulParses);
663     LOG.info("Names parse fail: " + parseFails);
664     LOG.info("Names parse errors: " + parseErrors);
665     LOG.info("Names ignoring authors: " + parseNoAuthors);
666     LOG.info("Total time: " + (end - start));
667     LOG.info("Average per name: " + (((double) end - start) / cnt));
668 
669     if ((parseFails + parseErrors) > CURRENTLY_FAIL) {
670       TestCase.fail(
671         "We are getting worse, not better. Currently failing: " + (parseFails + parseErrors) + ". We used to fail :"
672         + CURRENTLY_FAIL);
673     } else if (parseFails + parseErrors < CURRENTLY_FAIL) {
674       LOG.info("We are getting better! Used to fail : " + CURRENTLY_FAIL);
675     }
676     if ((parseFails + parseNoAuthors) > CURRENTLY_FAIL_EXCL_AUTHORS) {
677       TestCase.fail(
678         "We are getting worse, not better. Currently failing ignroing authors: " + (parseFails + parseNoAuthors)
679         + ". Was passing:" + CURRENTLY_FAIL_EXCL_AUTHORS);
680     } else if (parseFails + parseNoAuthors < CURRENTLY_FAIL_EXCL_AUTHORS) {
681       LOG.info("We are getting better! Used to fail without authors: " + CURRENTLY_FAIL_EXCL_AUTHORS);
682     }
683 
684   }
