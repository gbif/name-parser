232  /**
233   * Carefully normalizes a scientific name trying to maintain the original as close as possible.
234   * In particular the string is normalized by:
235   * - adding commas in front of years
236   * - trims whitespace around hyphens
237   * - unescapes unicode chars \\uhhhh, \\nnn, \xhh
238   * - pads whitespace around &
239   * - adds whitespace after dots following a genus abbreviation or rank marker
240   * - keeps whitespace before opening and after closing brackets
241   * - removes whitespace inside brackets
242   * - removes whitespace before commas
243   * - normalized hybrid marker to be the ascii multiplication sign
244   * - removes whitespace between hybrid marker and following name part in case it is NOT a hybrid formula
245   * - trims the string and replaces multi whitespace with single space
246   * - capitalizes all only uppercase words (authors are often found in upper case only)
247   *
248   * @param name To normalize
249   *
250   * @return The normalized name
251   */
252  public static String normalize(String name) {
253    if (name == null) {
254      return null;
255    }
256    name = org.gbif.utils.text.StringUtils.unescapeUnicodeChars(name);
257
258    // normalise usage of dots after abbreviated genus and rank marker
259    Matcher m = NORM_DOTS.matcher(name);
260    if (m.find()) {
261      name = m.replaceAll("$1. ");
262    }
263    // use commas before years
264    // ICZN §22A.2 http://www.iczn.org/iczn/includes/page.jsp?article=22&nfv=
265    m = COMMA_BEFORE_YEAR.matcher(name);
266    if (m.find()) {
267      name = m.replaceAll("$1, $2");
268    }
269    // no whitespace around hyphens
270    name = NORM_HYPHENS.matcher(name).replaceAll("-");
271    // use whitespace with &
272    name = NORM_AMPERSAND_WS.matcher(name).replaceAll(" & ");
273
274    // whitespace before and after brackets, keeping the bracket style
275    m = NORM_BRACKETS_OPEN.matcher(name);
276    if (m.find()) {
277      name = m.replaceAll(" $1");
278    }
279    m = NORM_BRACKETS_CLOSE.matcher(name);
280    if (m.find()) {
281      name = m.replaceAll("$1 ");
282    }
283    // remove whitespace before commas and replace double commas with one
284    m = NORM_COMMAS.matcher(name);
285    if (m.find()) {
286      name = m.replaceAll(", ");
287    }
288    // normalize hybrid markers
289    m = NORM_HYBRIDS_GENUS.matcher(name);
290    if (m.find()) {
291      name = m.replaceFirst("×$1");
292    }
293    m = NORM_HYBRIDS_EPITH.matcher(name);
294    if (m.find()) {
295      name = m.replaceFirst("$1 ×$2");
296    }
297    m = NORM_HYBRIDS_FORM.matcher(name);
298    if (m.find()) {
299      name = m.replaceAll(" × ");
300    }
301    // capitalize all entire upper case words
302    m = NORM_UPPERCASE_WORDS.matcher(name);
303    while (m.find()) {
304      name = name.replaceFirst(m.group(0), m.group(1) + m.group(2).toLowerCase());
305    }
306
307    // finally whitespace and trimming
308    name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
309    return StringUtils.trimToEmpty(name);
310  }

312  /**
313   * Does the same as a normalize and additionally removes all ( ) and "und" etc
314   * Checks if a name starts with a blacklisted name part like "Undetermined" or "Uncertain" and only returns the
315   * blacklisted word in that case
316   * so its easy to catch names with blacklisted name parts.
317   *
318   * @param name To normalize
319   *
320   * @return The normalized name
321   */
322  protected static String normalizeStrong(String name) {
323    if (name == null) {
324      return null;
325    }
326    // normalize all quotes to single "
327    name = NORM_QUOTES.matcher(name).replaceAll("'");
328    // enclosing quotes
329    name = REPLACE_QUOTES.matcher(name).replaceAll("");
330    // no question marks after words (after years they should remain!)
331    Matcher m = NO_Q_MARKS.matcher(name);
332    if (m.find()) {
333      name = m.replaceAll("$1");
334    }
335
336    // remove prefixes
337    name = NORM_PREFIXES.matcher(name).replaceAll("");
338
339    // remove brackets inside the genus, the kind taxon finder produces
340    m = NORM_TF_GENUS.matcher(name);
341    if (m.find()) {
342      name = m.replaceAll("$1$2 ");
343    }
344    // replace imprint years. See ICZN §22A.2.3 http://www.iczn.org/iczn/index.jsp?nfv=&article=22
345    // Ctenotus alacer Storr, 1970 ["1969"] -> Ctenotus alacer Storr, 1970
346    // C. Flahault 1887 ("1886-1888") -> C. Flahault 1887
347    m = NORM_IMPRINT_YEAR.matcher(name);
348    if (m.find()) {
349      // System.out.println(m.group(0));
350      // System.out.println(m.group(1));
351      name = m.replaceAll("$1");
352    }
353
354    // replace bibliographic in authorship
355    /*
356     * m = NORM_IN_BIB.matcher(name);
357     * if (m.find()) {
358     * // keep year if it only exists in IN reference
359     * Matcher mIN = EXTRACT_YEAR.matcher(m.group(0));
360     * name = m.replaceAll("");
361     * Matcher mNAME = EXTRACT_YEAR.matcher(name);
362     * if (mIN.find() && !mNAME.find()) {
363     * name = name + ", " + mIN.group(1);
364     * }
365     * }
366     */
367
368    // This is redundant, as it is done in the regular normalize function already
369    // BUT somehow all upper case authors slow down parsing so much that it can even come to an hold in the next step
370    // so we pay the price and do it twice
371    // capitalize all entire upper case words
372    m = NORM_UPPERCASE_WORDS.matcher(name);
373    while (m.find()) {
374      name = name.replaceFirst(m.group(0), m.group(1) + m.group(2).toLowerCase());
375    }
376
377    /*
378     * ICBN §46.2, Note 1.
379     * When authorship of a name differs from authorship of the publication in which it was validly published, both are
380     * sometimes cited,
381     * connected by the word "in". In such a case, "in" and what follows are part of a bibliographic citation and are
382     * better omitted
383     * unless the place of publication is being cited.
384     */
385
386    // normalise original name authorship, putting author AND year in brackets
387    // with long authorships this gets slow. Test if year exists first:
388    m = EXTRACT_YEAR.matcher(name);
389    if (m.find() && name.length() < 80) {
390      m = NORM_ORIG_AUTH.matcher(name);
391      if (m.find()) {
392        name = m.replaceAll("($1 $2)");
393      }
394      m = NORM_ORIG_AUTH2.matcher(name);
395      if (m.find()) {
396        name = m.replaceAll("($1 $2)");
397      }
398    }
399
400    // replace square brackets, keeping content (or better remove all within?)
401    name = NORM_NO_SQUARE_BRACKETS.matcher(name).replaceAll(" $1 ");
402    // replace different kind of brackets with ()
403    name = NORM_BRACKETS_OPEN_STRONG.matcher(name).replaceAll(" (");
404    name = NORM_BRACKETS_CLOSE_STRONG.matcher(name).replaceAll(") ");
405    // normalise different usages of ampersand, and, et &amp;
406    name = NORM_AND.matcher(name).replaceAll(" & ");
407    // but keep "et al." instead of "& al."
408    name = NORM_ET_AL.matcher(name).replaceAll(" et al.");
409
410    // // add commas between authors in space delimited list
411    // m = NORM_AUTH_DELIMIT.matcher(name);
412    // if (m.find()){
413    // name = m.replaceAll("$1, $2");
414    // }
415    // Bryozoan indet. 1
416    // Bryozoa sp. 2
417    // Bryozoa sp. E
418
419    name = NORM_SUFFIXES.matcher(name).replaceAll("");
420
421    // add parenthesis around subgenus if missing
422    m = NORM_SUBGENUS.matcher(name);
423    if (m.find()) {
424      name = m.replaceAll("$1 ($2) $3");
425    }
426
427    // finally whitespace and trimming
428    name = normalize(name);
429    // name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
430    return StringUtils.trimToEmpty(name);
431  }

433  /**
434   * basic careful cleaning, trying to preserve all parsable name parts
435   */
436  protected static String preClean(String name) {
437    // unescape unicode
438    name = org.gbif.utils.text.StringUtils.unescapeUnicodeChars(name);
439    // remove bad whitespace in html entities
440    Matcher m = XML_ENTITY_STRIP.matcher(name);
441    if (m.find()) {
442      name = m.replaceAll("&$1;");
443    }
444    // unescape html entities
445    name = StringEscapeUtils.unescapeHtml4(name);
446    // finally remove still existing bad ampersands missing the closing ;
447    name = AMPERSAND_ENTITY.matcher(name).replaceAll("& ");
448    // replace xml tags
449    name = XML_TAGS.matcher(name).replaceAll("");
450    // trim
451    name = name.trim();
452    // remove quotes in beginning and matching ones at the end
453    for (char c : QUOTES) {
454      int idx = 0;
455      while (idx < name.length() && (c == name.charAt(idx) || Character.isWhitespace(name.charAt(idx)))) {
456        idx++;
457      }
458      if (idx > 0) {
459        // check if we also find this char at the end
460        int end = 0;
461        while (c == name.charAt(name.length() - 1 - end) && (name.length() - idx - end) > 0) {
462          end++;
463        }
464        name = name.substring(idx, name.length() - end);
465      }
466    }
467    name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
468    return StringUtils.trimToEmpty(name);
469  }

471  /**
472   * Fully parse the supplied name also trying to extract authorships, a conceptual sec reference, remarks or notes
473   * on the nomenclatural status. In some cases the authorship parsing proves impossible and this nameparser will
474   * return null.
475   *
476   * For strings which are no scientific names and scientific names that cannot be expressed by the ParsedName class
477   * the parser will throw an UnparsableException with a given NameType and the original, unparsed name. This is the
478   * case for all virus names and proper hybrid formulas, so make sure you catch and process this exception.
479   *
480   * @param scientificName the full scientific name to parse
481   * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
482   *
483   * @throws org.gbif.nameparser.UnparsableException
484   */
485  public ParsedName parse(final String scientificName, @Nullable Rank rank) throws UnparsableException {
486    if (Strings.isNullOrEmpty(scientificName)) {
487      throw new UnparsableException(NameType.NO_NAME, scientificName);
488    }
489    long start = 0;
490    if (LOG.isDebugEnabled()) {
491      start = System.currentTimeMillis();
492    }
493
494    ParsedName pn = new ParsedName();
495    pn.setScientificName(scientificName);
496
497    // clean name, removing seriously wrong things
498    String name = preClean(scientificName);
499
500    // remove extinct markers
501    name = EXTINCT_PATTERN.matcher(name).replaceFirst("");
502
503    // before any cleaning test for properly quoted candidate names
504    Matcher m = IS_CANDIDATUS_QUOTE_PATTERN.matcher(scientificName);
505    if (m.find()) {
506      pn.setType(NameType.CANDIDATUS);
507      name = m.replaceFirst(m.group(2));
508    }
509
510    // normalize bacterial rank markers
511    name = TYPE_TO_VAR.matcher(name).replaceAll("$1var");
512
513    // parse out species/strain names with numbers found in Genebank/EBI names, e.g. Advenella kashmirensis W13003
514    m = STRAIN.matcher(name);
515    if (m.find()) {
516      name = m.replaceFirst(m.group(1));
517      pn.setType(NameType.INFORMAL);
518      pn.setStrain(m.group(2));
519      LOG.debug("Strain: {}", m.group(2));
520    }
521
522    // detect unparsable names
523    if (PLACEHOLDER.matcher(name).find()) {
524      throw new UnparsableException(NameType.PLACEHOLDER, scientificName);
525    }
526
527    if (IS_VIRUS_PATTERN.matcher(name).find() || IS_VIRUS_PATTERN_CASE_SENSITIVE.matcher(name).find()) {
528      throw new UnparsableException(NameType.VIRUS, scientificName);
529    }
530
531    // detect RNA/DNA gene/strain names and flag as informal
532    if (IS_GENE.matcher(name).find()) {
533      pn.setType(NameType.INFORMAL);
534    }
535
536    // normalise name
537    name = normalize(name);
538    if (Strings.isNullOrEmpty(name)) {
539      throw new UnparsableException(NameType.NO_NAME, scientificName);
540    }
541
542    // parse cultivar names first BEFORE we strongly normalize
543    // this will potentially remove quotes needed to find cultivar names
544    m = CULTIVAR.matcher(name);
545    if (m.find()) {
546      pn.setCultivarEpithet(m.group(1));
547      name = m.replaceFirst(" ");
548      pn.setType(NameType.CULTIVAR);
549      pn.setRank(Rank.CULTIVAR);
550      LOG.debug("Cultivar: {}", pn.getCultivarEpithet());
551    }
552
553    // name without any latin char letter at all?
554    if (NO_LETTERS.matcher(name).find()) {
555      throw new UnparsableException(NameType.NO_NAME, scientificName);
556    }
557
558    if (HYBRID_FORMULA_PATTERN.matcher(name).find()) {
559      throw new UnparsableException(NameType.HYBRID, scientificName);
560    }
561
562    m = IS_CANDIDATUS_PATTERN.matcher(name);
563    if (m.find()) {
564      pn.setType(NameType.CANDIDATUS);
565      name = m.replaceFirst("");
566    }
567
568    // extract nom.illeg. and other nomen status notes
569    m = EXTRACT_NOMSTATUS.matcher(name);
570    if (m.find()) {
571      pn.setNomStatus(StringUtils.trimToNull(m.group(1)));
572      name = m.replaceFirst("");
573      // if there was a rank given in the nom status populate the rank marker field
574      if (pn.getNomStatus() != null) {
575        Matcher rm = NOV_RANK_MARKER.matcher(pn.getNomStatus());
576        if (rm.find()) {
577          pn.setRankMarker(rm.group(1).trim());
578        }
579      }
580    }
581
582    // extract sec reference
583    m = EXTRACT_SENSU.matcher(name);
584    if (m.find()) {
585      pn.setSensu(StringUtils.trimToNull(m.group(1)));
586      name = m.replaceFirst("");
587    }
588    // extract other remarks
589    m = EXTRACT_REMARKS.matcher(name);
590    if (m.find()) {
591      pn.setRemarks(StringUtils.trimToNull(m.group(1)));
592      name = m.replaceFirst("");
593    }
594
595    // check for indets
596    m = RANK_MARKER_AT_END.matcher(name);
597    // f. is a marker for forms, but more often also found in authorships as "filius" - son of.
598    // so ignore those
599    if (m.find() && !(name.endsWith(" f.") || name.endsWith(" f"))) {
600      pn.setType(NameType.INFORMAL);
601      pn.setRankMarker(m.group(2));
602      name = m.replaceAll(" ");
603    }
604    m = NORM_INDET.matcher(name);
605    if (m.find()) {
606      pn.setType(NameType.INFORMAL);
607      name = m.replaceAll(" ");
608    }
609
610    name = normalizeStrong(name);
611    if (Strings.isNullOrEmpty(name)) {
612      throw new UnparsableException(NameType.DOUBTFUL, scientificName);
613    }
614
615    // remember current rank for later reuse
616    Rank origRank = pn.getRank();
617
618    // try regular parsing
619    boolean parsed = nnParser.parseNormalisedName(pn, name, rank);
620    if (!parsed) {
621      // try again with stronger normalisation, maybe that helps...
622      LOG.debug("Can't parse, use dirty normalizer");
623      final String deDirtedName = cleanStrong(name);
624      parsed = nnParser.parseNormalisedName(pn, deDirtedName, rank);
625      if (!parsed) {
626        LOG.debug("Still can't parse, try to ignore authors");
627        // try to parse canonical alone ignoring authorship as last resort
628        parsed = nnParser.parseNormalisedNameIgnoreAuthors(pn, deDirtedName, rank);
629        pn.setAuthorsParsed(false);
630        if (!parsed) {
631          // we just cant parse this one
632          // try to spot a virus name once we know its not a scientific name
633          m = IS_VIRUS_PATTERN_POSTFAIL.matcher(name);
634          if (m.find()) {
635            throw new UnparsableException(NameType.VIRUS, scientificName);
636          }
637
638          throw new UnparsableException(NameType.DOUBTFUL, scientificName);
639        }
640      }
641    }
642    if (origRank != null) {
643      pn.setRank(origRank);
644    }
645
646    // primarily make sure again we haven't parsed a non name
647    postAssertParsing(pn, scientificName, name);
648
649    // determine name type if not yet assigned
650    if (pn.getType() == null) {
651      // a doubtful name?
652      m = DOUBTFUL.matcher(scientificName);
653      if (!m.find()) {
654        pn.setType(NameType.DOUBTFUL);
655      } else {
656        pn.setType(NameType.SCIENTIFIC);
657      }
658    }
659
660    LOG.debug("Parsing time: {}", (System.currentTimeMillis() - start));
661    return pn;
662  }
663
664  private void postAssertParsing(ParsedName pn, final String rawName, final String normedName) throws UnparsableException {
665    // if we only match a monomial in the 3rd pass its suspicious
666    if (pn.getGenusOrAbove() != null && !pn.isBinomial() && Character.isLowerCase(normedName.charAt(0))) {
667      // a monomial match, but it was a lower case name - doubtful at least!
668      throw new UnparsableException(NameType.DOUBTFUL, rawName);
669
670    } else if (pn.getRank() != null) {
671      if (pn.getRank().isSpeciesOrBelow() && !pn.isBinomial() && !pn.getRank().equals(Rank.CULTIVAR)) {
672        pn.setType(NameType.INFORMAL);
673      } else if (pn.getRank().equals(Rank.CULTIVAR) && pn.getCultivarEpithet() == null) {
674        pn.setType(NameType.INFORMAL);
675      } else if (pn.getRank().isInfraspecific() && pn.getInfraSpecificEpithet() == null) {
676        pn.setType(NameType.INFORMAL);
677      } else if (!pn.getRank().isSpeciesOrBelow() && pn.isBinomial()) {
678        pn.setType(NameType.DOUBTFUL);
679      }
680    }
681  }

683  /**
684   * parses the name without authorship and returns the ParsedName.canonicalName() string
685   * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
686   */
687  public String parseToCanonical(String scientificName, @Nullable Rank rank) {
688    if (Strings.isNullOrEmpty(scientificName)) {
689      return null;
690    }
691    try {
692      ParsedName pn = parse(scientificName, rank);
693      if (pn != null) {
694        return pn.canonicalName();
695      }
696    } catch (UnparsableException e) {
697      LOG.warn("Unparsable name " + scientificName + " >>> " + e.getMessage());
698    }
699    return null;
700  }
